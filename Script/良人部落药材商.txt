procedure OnLeftClick (aStr : String);
var
   Str : String;
   Race : Integer;
begin
   Str := callfunc ('getsenderrace');
   Race := StrToInt (Str);
   if Race = 1 then begin
      Str := 'showwindow .\help\quest药材商6.txt 1';
      print (Str);
      exit; 
   end;
end;

procedure OnGetResult (aStr : String);
var
   Str, Name : String;
begin
   if aStr = 'close' then begin
      exit;
   end;
   // 剧牢何遏
   if aStr = 'sendmaterial' then begin
      Str := callfunc ('getsenderitemexistence 药材商材料书札:1');
      if Str = 'false' then begin
         print ('say 给衣服染上色吧..');
         exit;
      end;
   
      Str := callfunc ('getsenderitemexistence 药材商材料3:1');
      if Str = 'false' then begin
         print ('showwindow .\help\quest药材商4_6.txt 1');
         exit;
      end;
      print ('showwindow .\help\quest药材商4_5.txt 0');
      exit;
   end;
   if aStr = 'q2_3_close' then begin
      Str := callfunc ('checkenoughspace');
      if Str = 'false' then begin
         print ('say 物品栏已满...');
         exit;
      end;
      Str := callfunc ('getsenderitemexistence 药材商材料3:1');
      if Str = 'false' then begin
         print ('say 给衣服染上色吧..');
         exit;
      end;

      print ('getsenderitem 药材商材料3:1');
      exit;
   end;
   if aStr = 'continue' then begin
      exit;
   end;
   if aStr = 'giveup' then begin
      print ('getsenderitem 药材商材料书札:1');
      print ('getsenderitem 药材商材料1:1');
      print ('getsenderitem 药材商材料2:1');
      print ('getsenderitem 药材商材料3:1');                  
      exit;
   end;
end;
